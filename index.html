# Unir ambas tablas por la columna Id y la fecha

Tabla_combinada <- merge(Actividad.Diaria, Sueño, by.x = c("Id", "ActivityDate"),

by.y = c("Id", "Date"))

# Visualizar la relación entre pasos y minutos de sueño

ggplot(Tabla_combinada, aes(x = TotalSteps, y = TotalMinutesAsleep, color = TotalSteps)) +

geom_point(size = 3, alpha = 0.7) + # puntos más grandes y transparentes

geom_smooth(method = "lm", color = "red", fill = "blue", alpha = 0.3) + # línea de regresión con colores

scale_color_gradient(low = "yellow", high = "red") + # degradado de color para los puntos

labs(title = "Relación entre los pasos diarios y minutos de sueño",

x = "Pasos totales",

y = "Minutos totales de sueño") +

theme_minimal() +

theme(plot.title = element_text(color = "hotpink", size = 18),

axis.title.x = element_text(color = "darkgreen"),

axis.title.y = element_text(color = "darkred"))

# Calcular el promedio de pasos por día de la semana

steps_by_day <- Actividad.Diaria %>%

group_by(DayOfWeek) %>%

summarise(AverageSteps = mean(TotalSteps))

# Reordenar los días de la semana

steps_by_day$DayOfWeek <- factor(steps_by_day$DayOfWeek,

levels = c("Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"))

# Gráfico de barras del promedio de pasos por día

ggplot(steps_by_day, aes(x = DayOfWeek, y = AverageSteps, fill = DayOfWeek)) +

geom_bar(stat = "identity", color = "black", alpha = 0.7) +

scale_fill_viridis_d() +

labs(title = "Promedio de pasos por día de la semana",

x = "Día de la semana",

y = "Pasos promedio") +

theme_minimal()+

theme(plot.title = element_text(color = "hotpink", size = 16),

axis.title.x = element_text(color = "darkgreen"),

axis.title.y = element_text(color = "darkred"))
